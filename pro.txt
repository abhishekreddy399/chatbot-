    <script>
        // DOM Elements
        const heroCta = document.getElementById('hero-cta');
        const navTryNow = document.getElementById('nav-try-now');
        const chatbotContainer = document.getElementById('chatbot-container');
        const chatMessages = document.getElementById('chatbot-messages');
        const userInput = document.getElementById('user-input');
        const sendButton = document.getElementById('send-button');

        // Show chatbot when CTA buttons are clicked
        heroCta.addEventListener('click', showChatbot);
        navTryNow.addEventListener('click', showChatbot);

        function showChatbot(e) {
            e.preventDefault();
            chatbotContainer.style.display = 'block';
            // Scroll to chatbot
            chatbotContainer.scrollIntoView({ behavior: 'smooth' });
        }

        // Handle user input
        sendButton.addEventListener('click', handleUserInput);
        userInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                handleUserInput();
            }
        });

        // Sample order data (in a real app, this would come from a database)
        const sampleOrders = {
            '12345': {
                customer: 'John Doe',
                status: 'Shipped',
                items: ['Wireless Headphones', 'USB-C Cable'],
                trackingNumber: 'UPS-789456123',
                estimatedDelivery: 'April 10, 2023'
            },
            '67890': {
                customer: 'Jane Smith',
                status: 'Processing',
                items: ['Smart Watch', 'Screen Protector'],
                trackingNumber: null,
                estimatedDelivery: 'April 15, 2023'
            }
        };

        function handleUserInput() {
            const inputText = userInput.value.trim();
            if (!inputText) return;

            // Add user message to chat
            addMessage(inputText, 'user');

            // Clear input
            userInput.value = '';

            // Process input after a short delay to simulate thinking
            setTimeout(() => {
                processOrderRequest(inputText);
            }, 500);
        }

        function addMessage(text, sender) {
            const messageDiv = document.createElement('div');
            messageDiv.classList.add('message');
            messageDiv.classList.add(sender === 'user' ? 'user-message' : 'bot-message');
            messageDiv.textContent = text;
            chatMessages.appendChild(messageDiv);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        function processOrderRequest(input) {
            // Check if input matches an order number pattern
            const orderNumberMatch = input.match(/#?(\d{5})/);
            
            if (orderNumberMatch) {
                const orderNumber = orderNumberMatch[1];
                if (sampleOrders[orderNumber]) {
                    const order = sampleOrders[orderNumber];
                    let response = `Order #${orderNumber} for ${order.customer}:\n`;
                    response += `Status: ${order.status}\n`;
                    response += `Items: ${order.items.join(', ')}\n`;
                    
                    if (order.trackingNumber) {
                        response += `Tracking Number: ${order.trackingNumber}\n`;
                    }
                    
                    response += `Estimated Delivery: ${order.estimatedDelivery}`;
                    
                    addMessage(response, 'bot');
                } else {
                    addMessage(`I couldn't find order #${orderNumber}. Please check the number and try again.`, 'bot');
                }
            } 
            // Check if input contains a customer name
            else if (input.toLowerCase().includes('john') || input.toLowerCase().includes('doe')) {
                addMessage("I found an order for John Doe:\nOrder #12345 is shipped and should arrive by April 10.", 'bot');
            } 
            else if (input.toLowerCase().includes('jane') || input.toLowerCase().includes('smith')) {
                addMessage("I found an order for Jane Smith:\nOrder #67890 is processing and should ship soon.", 'bot');
            }
            else {
                addMessage("I'm not sure I understand. Please provide an order number or customer name. For example: 'Tracking order #12345' or 'Where is my order for John Doe?'", 'bot');
            }
        }
    </script>
